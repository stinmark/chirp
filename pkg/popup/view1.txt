package popup

import (
	_ "embed" // Required for embedding files

	"github.com/stinmark/chirp/pkg/helpers"
	"github.com/stinmark/chirp/pkg/theme"

	tea "charm.land/bubbletea/v2"
	"charm.land/lipgloss/v2"
)

//go:embed bird.txt
var birdArt string

//go:embed banner.txt
var bannerText string

func (m PopupModel) View() tea.View {
	// 1. declare box width
	// 2. Render your layout elements
	artStyled := lipgloss.NewStyle().Foreground(lipgloss.Color("4")).Bold(true).Render(birdArt)
	//bannerTitle := theme.GenerateTexturedShadowTitle(theme.TruncateString("CHIRP", 10, false), "1")

	var middlePanel []string
	middlePanel = append(middlePanel, artStyled)
	middlePanel = append(middlePanel, theme.AccentStyle.Width(40).AlignHorizontal(lipgloss.Right).Render(m.Chirp.Message)+"\n")

	var bottomPanel []string
	bottomPanel = append(bottomPanel, theme.MutedStyle.Render("daemon "+helpers.Ternary(m.DaemonRunning, "● Running", "● stopped")))

	if m.Chirp.AutoRepeat {
		bottomPanel = append(bottomPanel, theme.ActiveStye.Render("mode: AutoRepeat"))
		bottomPanel = append(bottomPanel, theme.MutedStyle.Render("[r] stop repeat [s/q] quit"))
	} else {
		bottomPanel = append(bottomPanel, theme.ActiveStye.Render("mode: Run Once"))
		bottomPanel = append(bottomPanel, theme.MutedStyle.Render("[r] repeat [s/q] quit"))
	}

	// 3. Join layout elements together Dynamically

	// --- STEP 3A: Render the core middle content to measure its true width ---
	middleView := lipgloss.JoinHorizontal(lipgloss.Center, middlePanel...)
	contentWidth := lipgloss.Width(middleView)

	// --- STEP 3B: Use that exact width to stretch your banner and bottom panel ---
	bannerStyle := lipgloss.NewStyle().
		Foreground(lipgloss.Color("1")).
		Border(lipgloss.NormalBorder(), false, false, true, false). // Bottom border only
		Width(contentWidth).                                        // Fills exact width of content
		AlignHorizontal(lipgloss.Center)

	bannerView := bannerStyle.Render(bannerText)

	bottomStyle := lipgloss.NewStyle().
		Border(lipgloss.NormalBorder(), true, false, false, false). // Top border only
		BorderForeground(lipgloss.Color("8")).
		Width(contentWidth). // Fills exact width of content
		AlignHorizontal(lipgloss.Center)

	bottomView := bottomStyle.Render(lipgloss.JoinHorizontal(lipgloss.Center, bottomPanel...))

	// --- STEP 3C: Stack them vertically and wrap in the main outer box ---
	stackedLayout := lipgloss.JoinVertical(
		lipgloss.Center,
		bannerView,
		middleView,
		bottomView,
	)

	combinedView := lipgloss.NewStyle().
		Border(lipgloss.NormalBorder()).
		BorderForeground(lipgloss.Color("8")).
		Padding(2).
		Render(stackedLayout)

	// 4. DYNAMICALLY CENTER THE CONTENT
	var finalRender string
	if m.TerminalWidth > 0 && m.TerminalHeight > 0 {
		finalRender = lipgloss.Place(
			m.TerminalWidth,
			m.TerminalHeight,
			lipgloss.Center,
			lipgloss.Center,
			combinedView,
		)
	} else {
		finalRender = lipgloss.NewStyle().Padding(2, 4).Render(combinedView)
	}

	v := tea.NewView(finalRender)
	v.AltScreen = true
	return v
}
